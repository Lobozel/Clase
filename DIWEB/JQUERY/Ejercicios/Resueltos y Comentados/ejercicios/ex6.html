¿Qué solución aporta este ejercicio?

Manejo de listas con JQuery.

  • Primer item.
  • Segundo item.
  • Tercer item.
  • Cuarto item.








<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script>
<script type="text/javascript">
$(function() {
//Asignamos el evento onclick a los botones
    $("#boton1").click(eliminarElementos);
   $("#boton2").click(restaurarLista);
   $("#boton3").click(anadirElementoFinal);
   $("#boton4").click(anadirElementoPrincipio);
   $("#boton5").click(eliminarElementoFinal);
   $("#boton6").click(eliminarElementoPrincipio);
   $("#boton7").click(decorar);
});
function eliminarElementos() {
//Seleccionamos los elementos ul y los declaramos vacios
//como solo tenemos una etiqueta ul (una lista), no hay problema, ¡¡pero si hubiese más hay que tener cuidado!!
    $("ul").empty();
}
function restaurarLista() {
//Cambiamos el código html de las etiquetas ul por la lista que teniamos al principio, preescrita a mano
    $("ul").html(
      "<li>Primer item.</li><li>Segundo item.</li><li>Tercer item.</li><li>Cuarto item.</li>"
   );
}
function anadirElementoFinal() {
//el método .append nos permite añadir texto con etiquetas html al final del contenido de la etiqueta/s que seleccionemos
    $("ul").append("<li>otro item al final</li>");
}
function anadirElementoPrincipio() {
//el método .prepend nos permite añadir texto con etiquetas html al principio del contenido de la etiqueta/s que seleccionemos
    $("ul").prepend("<li>otro item al principio</li>");
}
function eliminarElementoFinal() {
   var y;
   var cantidad = $("li").length; //recogemos la cantidad de etiquetas li hay (si tuviesemos más listas, tendriamos que especificar mejor aqui)
    y = $("li").eq(cantidad - 1); //el índice de la última es la cantidad -1 porque empieza a contar por 0
    y.remove(); //ahora que hemos localizado la última, la borramos
//también se podría hacer con $(li:last-child)
//$("li:last-child").remove();
 }
function eliminarElementoPrincipio() {
   $("li").eq(0).remove(); //seleccionamos el elemento li con el índice 0 y lo eliminamos
//también funcionaría: $('li:first-child').remove();
 }
function decorar() {
//Le cambia el estilo background-color de su css por el color #eee a los elementos li pares
    $("li:odd").css("background-color", "#eee");
//Le cambia el estilo background-color de su css por el color #ccc a los elementos li impares
    $("li:even").css("background-color", "#ccc");
}
</script>
