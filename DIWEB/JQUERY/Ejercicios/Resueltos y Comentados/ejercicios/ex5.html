<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script>
<script type="text/javascript">
//-----SE QUE ES MUCHO, PERO ES BIEN SENCILLO-----//
//Se le asigna la función onclick a cada boton con su respectiva función
 $(function() {
   $("#html").click(fhtml);
   $("#val").click(fval);
   $("#text").click(ftext);
   $("#attr").click(fattr);
   $("#html2").click(fhtml2);
   $("#val2").click(fval2);
   $("#text2").click(ftext2);
   $("#attr2").click(fattr2);
   $("#html3").click(fhtml3);
   $("#val3").click(fval3);
   $("#text3").click(ftext3);
   $("#attr3").click(fattr3);
   $("#html4").click(fhtml4);
   $("#val4").click(fval4);
   $("#text4").click(ftext4);
   $("#attr4").click(fattr4);
})
//------------------------------------------------------------//
//------ALERTS DEL ELEMENTO CON EL ID capa (div)
 function fhtml()
//Muestra en un alert el código html dentro de la etiqueta con el id #capa
    {alert($("#capa").html());}
function ftext()
//Muestra en un alert el texto dentro de la etiqueta con el id #capa
    {alert($("#capa").text());}
function fval()
/*Mostraría el valor de la etiqueta con el id #capa, pero este no existe
por lo que no muestra nada*/
    {alert($("#capa").val());}
function fattr()
/*Muestra en un alert el valor del atributo id de la etiqueta con el id #capa
quizás no sea el mejor ejemplo...*/
    {alert($("#capa").attr("id"));}
//------------------------------------------------------------//
//------ALERTS DEL ELEMENTO CON EL ID cuadro (input [type="text"])
 function fhtml2()
/*Mostraría el texto, con etiquetas html, dentro de la etiqueta,
pero al ser un input, la etiqueta no lleva texto*/
    {alert($("#cuadro").html());}
function ftext2()
/*Mostraría el texto dentro de la etiqueta, pero al ser un input,
la etiqueta no lleva texto*/
    {alert($("#cuadro").text());}
function fval2()
//Muestra en un alert el valor actual de la etiqueta (el texto dentro del input vamos)
    {alert($("#cuadro").val());}
function fattr2()
/*Muestra en un alert el valor del atributo value de la etiqueta
no importa si se cambia el texto dentro de la caja, devolverá el valor inicial del atributo value*/
    {alert($("#cuadro").attr("value"));}
//------------------------------------------------------------//
//------CAMBIAR EL CONTENIDO DEL ELEMENTO CON EL ID capa (div)
 function fhtml3()
//Cambia el código html dentro de la etiqueta con el id capa por "Nuevo Contenido html"
    {$("#capa").html("Nuevo Contenido html");}
function ftext3()
//Cambia el texto de la etiqueta con el id capa por "Nuevo texto"
    {$("#capa").text("Nuevo texto");}
function fval3()
//Cambiaría el valor de la etiqueta, pero al no ser una etiqueta de formulario, no posee tal caracteristica
    {$("#capa").val("Nuevo valor");}
function fattr3()
//Cambia el valor del atributo id por Nuevoid, trás esto las demás funciones de #capa dejan de funcionar
//es debido a que todas buscan un elemento con el id capa, pero al cambiar este valor ya no existe ninguno con ese id
    {$("#capa").attr("id","Nuevoid");}
//------------------------------------------------------------//
//------CAMBIAR EL CONTENIDO DEL ELEMENTO CON EL ID cuadro (input [type="text"])
 function fhtml4()
//Cambiaría el código html de la etiqueta, pero al ser un campo de formulario, no contiene texto
    {$("#cuadro").html("Nuevo Contenido html");}
function ftext4()
//Cambiaría el texto de la etiqueta, pero al ser un campo de formulario, no contiene texto
    {$("#cuadro").text("Nuevo texto");}
function fval4()
//Cambia el contenido (valor) del cuadro de texto por "Nuevo valor"
    {$("#cuadro").val("Nuevo valor");}
function fattr4()
//Cambia el valor del atributo value por "Nuevo valor atributo"
//(Los cambios no se reflejan para el usuario, el texto dentro de la caja no varia)
    {$("#cuadro").attr("value","Nuevo valor atributo");}
//------------------------------------------------------------//
 </script>
