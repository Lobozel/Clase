Al hacer doble click sobre el <h1> de abajo, cambiará su clase con efectos visibles en su estilo.

Código CSS

<style>
   .box{
      background-color: sandybrown;
      color: #fff;
      border: 3px solid black;
   }
   .otherClass{ /*lo se, me falta imaginación*/
      letter-spacing: 0.3em;
      font-size: small;
   }
</style>
<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script>
<script type="text/javascript">
$(function(){
//Seleccionamos las etiquetas <h1> con la clase "box" o la clase "otherClass"
//y le asignamos el evento dblclick que se ejecuta al hacer doble click sobre el elemento.
   $("h1.box, h1.otherClass").dblclick(changeClass);
})
function changeClass(){
/*Si el elemento con el que se llama al método tiene la clase "box",
le quita la clase "box" y le añade la clase "otherClass"*/
   if($(this).hasClass("box")){
      $(this).removeClass("box").addClass("otherClass");
   }else{
//En caso contrario, la clase es "otherClass", se le quita dicha clase y se le añade la clase "box"
      $(this).removeClass("otherClass").addClass("box");
   }
}
</script>
