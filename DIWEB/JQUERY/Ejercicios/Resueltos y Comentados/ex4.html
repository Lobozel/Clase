<script type="text/javascript">
$(function() {
   $("#ver1").click(ver1);
   $("#ocultar1").click(ocultar1);
   $("#ver2").click(ver2);
   $("#ocultar2").click(ocultar2);
   $("#ver3").click(ver3);
   $("#ocultar3").click(ocultar3);
   $("#verocultar").click(verOcultar);
   $("#animar").click(animar);
})
function ver1() {
//muestra la tabla de forma rápida, en unos 200 milisegundos
    $("#tabla").show("fast");
}
function ocultar1() {
//oculta la tabla de forma lenta, en unos 600 milisegundos
    $("#tabla").hide("slow");
}
function ver2() {
/*muestra la tabla de forma rápida, en unos 200 milisegundos
utiliza la animacion fade que da un efecto de desvanecerse*/
    $("#tabla").fadeIn("fast");
}
function ocultar2() {
/*oculta la tabla de forma lenta, en unos 600 milisegundos
utiliza la animacion fade que da un efecto de desvanecerse*/
    $("#tabla").fadeOut("slow");
}
function ver3() {
/*muestra la tabla de forma lenta, en unos 600 milisegundos
utiliza la animacion fade que da un efecto de desvanecerse
el segundo parámetro indica la opacidad, 1 significa una opacidad del 100%*/
    $("#tabla").fadeTo("slow", 1);
}
function ocultar3() {
/*muestra la tabla de forma lenta, en unos 600 milisegundos
utiliza la animacion fade que da un efecto de desvanecerse
el segundo parámetro indica la opacidad, 0.5 significa una opacidad del 50%*/
    $("#tabla").fadeTo("slow", 0.5);
}
function verOcultar(){
/*El método toggle oculta o muestra el elemento
el parámetro slow indica que queremos que la animación sea lenta, de unos 600 milisegundos*/
    $("#tabla").toggle("slow");
}
function animar() {
/*crea una animación, parecida a la que usabamos en CSS con transition
agranda el tamaño de la letra y el tamaño del borde inferior*/
    $("#titulo").animate({
      "border-bottom-width": "20",
      "font-size": "25pt"
   });
   }
</script>
